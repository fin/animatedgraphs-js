1 /**
  2  * <p>AnimatedGraphs:</p>
  3  *
  4  * <p>graphs & animations between them.</p>
  5  * <p>wraps raphael.js</p>
  6  *
  7  *
  8  * <pre>datasets = array of objects, e.g. [{male: {asian: 1, african: 2}, female: {asian: 3, african: 2}}]</pre>
  9  * <pre>labels = labels for objects, e.g. ['gender', 'ethnicity']</pre>
 10  *
 11  *
 12  * @class
 13  */
 14 function AG(/** [{…key:value…}] */ datasets, /** [String] */ labels) {
 15     
 16     var self /** @lends AG# */  = this;
 17 
 18     self.datasets = datasets;
 19     self.datasets_raw = [];
 20     self.labels = labels;
 21 
 22     self.dataset = null;
 23     self.dataset_raw = null;
 24     self.current_chart_type = null;
 25     self.current_elements = null;
 26     self.size_x = 700;
 27     self.size_y = 450;
 28 
 29     self.raphael = new Raphael("testcanvas");
 30     var raphael = self.raphael;
 31     var width = raphael.width;
 32     var height = raphael.height;
 33 
 34 
 35     self.chart_functions =  {
 36                     'bar': function(values) { return raphael.g.barchart_paths(0,0,width,height,values, {barwidth: 40, to: 70, stretch: false }); },
 37                     'bar_grouped': function(values) { return raphael.g.barchart_paths(0,0,width,height,values, {barwidth: 40, gutter: '100%', to: 70, stretch: false}); },
 38                     'bar_stacked': function(values) { return raphael.g.barchart_paths(0,0,width,height,values, {stacked: true, barwidth: 40, gutter: '100%', to: 70, stretch: false}); },
 39                     'pie': function(values) { return raphael.g.ringchart_paths(0,0,width,height,values); },
 40                     'pie_separate': function(values) { return raphael.g.ringchart_paths(0,0,width,height,values,{separate: true}); }
 41                 };
 42 
 43 
 44     /**
 45      * switch dataset to index. need to call .chart() to take effect!
 46      * @memberOf AG#
 47      * @public
 48      */
 49     self.set_dataset = function(/** int */ id) /** void */{
 50         self.dataset = self.datasets[id];
 51         self.dataset_raw = self.datasets_raw[id];
 52     };
 53 
 54     /* 
 55      * annotate
 56      * prepend object keys in datasets with their respective labels
 57      * (for unique identification later on)
 58      * @function
 59      */
 60     /** @memberOf AG#
 61      * @name annotate_datasets
 62      * @private
 63      */
 64     self.annotate_datasets = function() {
 65         for(var i=0;i<self.datasets.length;i++) {
 66             self.annotate_dataset(self.datasets[i]);
 67         }
 68     };
 69     /** @memberOf AG#
 70      *  @name annotate_dataset
 71      *  @private
 72      */
 73     self.annotate_dataset = function(dataset) {
 74         var annotate = function(input, level) {
 75             for(var k in input) {
 76                 if(!input.hasOwnProperty(k))
 77                     continue;
 78                 annotate(input[k], level+1);
 79                 input[self.labels[level]+':'+k] = input[k];
 80                 delete input[k];
 81             }
 82         };
 83         annotate(dataset, 0, {});
 84     };
 85 
 86     /*
 87      * summarize
 88      * basically denormalize datasets in this.datasets_raw
 89      * enables easier filtering and transformations later on
 90      */
 91     /** @memberOf AG#
 92      *  @name summarize_datasets
 93      *  @private
 94      */
 95     self.summarize_datasets = function() {
 96         for(var i=0;i<self.datasets.length;i++) {
 97             self.summarize_dataset(i);
 98         }
 99     };
100     /** @memberOf AG#
101      *  @name summarize_dataset
102      *  @private
103      */
104     self.summarize_dataset = function(dataset_index) {
105         var dataset = self.datasets[dataset_index];
106         self.datasets_raw.push(self.summarize_dataset_recurse(dataset, [],0));
107     };
108     /** @memberOf AG#
109      *  @name summarize_dataset_recurse
110      *  @private
111      */
112     self.summarize_dataset_recurse = function(obj, current_key, depth) {
113         var result = {};
114         if(Raphael.is(obj, 'object')) {
115             for(var key in obj) {
116                 if(!obj.hasOwnProperty(key))
117                     continue;
118                 var val = obj[key];
119                 var ck = current_key.clone();
120                 ck[self.labels[depth]]=key;
121                 result = result.merge(self.summarize_dataset_recurse(val, ck, depth+1));
122             }
123             return result;
124         } else {
125             result[current_key.jsonproper()] = obj;
126             return result;
127         }
128     };
129 
130     /** @memberOf AG#
131      *  @name match_elements
132      *  @private
133      */
134     self.match_elements = function(key, labels, elements) {
135         var matching_elements = [];
136         for(var j=0;j<elements.length;j++) {
137             var element = elements[j];
138             var ekey = element.key;
139             for(var k=0;k<labels.length;k++) {
140                 var label = labels[k];
141                 if(!ekey[label] || ekey[label] != key[label]) {
142                     break;
143                 }
144                 if(k==(labels.length-1)) {
145                     matching_elements.push(element);
146                 }
147             }
148         }
149         return matching_elements;
150     };
151 
152     /*
153      * draw/change chart
154      *
155      * e.g.
156      * chart('bar', ['gender']);
157      * or
158      * chart('bar-grouped', ['gender', 'ethnicity'])
159      */
160      /**
161      *  The workhorse. Draw (with animation from a current chart, if valid) a chart using the current dataset
162       * @methodOf AG#
163      *  @public
164      *
165      */
166     self.chart = function(/** String */ chart_type, /** [String] */labels) /** void */ {
167         var old_chart_type = self.current_chart_type;
168         var old_elements = self.current_elements;
169         self.current_elements = [];
170 
171         var data = self.get_data(labels);
172 
173         var chart_function = self.chart_functions[chart_type];
174 
175         var g = chart_function(data);
176         var max_zindex=0;
177         g.elements = flatten(g.elements);
178         if(old_elements) {
179             for(var i=0;i<g.elements.length;i++) {
180                 var element = g.elements[i];
181                 var matching_elements = self.match_elements(element.key, labels, old_elements);
182                 if(matching_elements.length>0) {
183                     element.raphael_element = matching_elements[0].raphael_element;
184                     var ndx = old_elements.indexOf(matching_elements[0]);
185                     if(ndx>=0) {
186                         old_elements.splice(ndx,1);
187                     } else {}
188                 } else {
189                     element.raphael_element = raphael.path(element.path);
190                 }
191                 element.raphael_element.animate({'path': element.path}, 1024);
192                 //element.raphael_element.attr(element.attr);
193                 self.current_elements.push(element);
194                 if(element.zindex && max_zindex<element.zindex)
195                     max_zindex = element.zindex;
196             }
197             for(var i=0;i<old_elements.length;i++) {
198                 old_elements[i].raphael_element.remove();
199             }
200         } else {
201             for(var i=0;i<g.elements.length;i++) {
202                 g.elements[i].raphael_element = raphael.path(g.elements[i].path);
203                 g.elements[i].raphael_element.attr(g.elements[i].attr);
204                 if(g.elements[i].zindex && max_zindex<g.elements[i].zindex) {
205                     max_zindex = g.elements[i].zindex;
206                 }
207                 self.current_elements.push(g.elements[i]);
208             }
209             self.current_chart_type = chart_type;
210         }
211         for(var zi=max_zindex-1;zi>=0;zi--) {
212             var es = self.current_elements.filter(function(x) {return x.zindex==zi;});
213             for(var i=0;i<es.length;i++) {
214                 es[i].raphael_element.toFront();
215             }
216         }
217     };

221     /** @methodOf AG#
222      *  @name get_data
223      *  @private
224      * get data formatted for g.raphael-like input
225      */
226     self.get_data = function(/** [String] */ labels) /** [{key:, value:}] */{
227         var dataset = self.dataset_raw;
228         var raw_values = [];
229         var lbls = labels;
230 
231         var tmp = [];
232 
233         var label_values = [];
234 
235         for(var i=0;i<labels.length;i++) {
236             var label = labels[i];
237             label_values.push([]);
238             for(var key in dataset) {
239                 if(!dataset.hasOwnProperty(key))
240                     continue;
241                 var k = JSON.parse(key)[label];
242                 if(label_values[i].indexOf(k)<0) {
243                     label_values[i].push(k);
244                 }
245                 //k = k.substring(label.length+1);
246                 //alert(k.substring(0));
247                 raw_values.push({key: JSON.parse(key), value: dataset[key]});
248                 //raw_values.push({key: k.substring(0), value: dataset[key]});
249             }
250         }
251 
252         /**
253          * @inner
254          * @ignore
255          */
256         var recurse = function(values, index, labels, label_values) {
257             var result = [];
258             for(var i=0;i<label_values[index].length;i++) {
259                 var lv = label_values[index][i];
260                 var fv = values.filter(function(x) { return x.key[labels[index]]==lv;  });
261                 if(index+1<labels.length) {
262                     fv = recurse(fv, index+1, labels, label_values);
263                     result.push(fv);
264                 } else {
265                     if(fv.length>0) {
266                         var s = 0;
267                         var k = {};
268                         var rl = "";
269                         for(var j=0;j<fv.length;j++) {
270                             s+=fv[j].value;
271                         }
272                         for(var j=0;j<labels.length;j++) {
273                             var l = labels[j];
274                             rl = l.length;
275                             k[l] = fv[0].key[l];
276                         }
277                         var re = k[l].substring(rl+1);
278                         result.push({key: k, label: re, value: s});
279                     }
280                 }
281             }
282             return result;
283         };
284         return recurse(raw_values, 0, labels, label_values);
285     };
286 
287 
288     self.annotate_datasets();
289     self.summarize_datasets();
290 
291     self.set_dataset(0);
292 }

296 Object.prototype.clone = function() {
297   var newObj = (this instanceof Array) ? [] : {};
298   for (i in this) {
299     if (i == 'clone') continue;
300     if (this[i] && typeof this[i] == "object") {
301       newObj[i] = this[i].clone();
302     } else newObj[i] = this[i]
303   } return newObj;
304 };

306 /** @ignore */
307 Object.prototype.jsonproper = function() {
308     var keys = [];
309     for(var key in this) {
310         if(!this.hasOwnProperty(key))
311             continue;
312         keys.push(key);
313     }
314     keys.sort();
315     var result = {};
316     for(var i=0;i<keys.length;i++) {
317         result[keys[i]] = this[keys[i]];
318     }
319     return JSON.stringify(result);
320 };

324 /** @ignore */
325 function flatten(array){
326     var flat = [];
327     for (var i = 0, l = array.length; i < l; i++){
328         var type = Object.prototype.toString.call(array[i]).split(' ').pop().split(']').shift().toLowerCase();
329         if (type) { flat = flat.concat(/^(array|collection|arguments)$/.test(type) ? flatten(array[i]) : array[i]); }
330     }
331     return flat;
332 }
