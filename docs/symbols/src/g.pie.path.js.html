 36         function sector(cx, cy, r, startAngle, endAngle, fill)
 37         {
 38             var rad = Math.PI / 180,
 39                 /*x1 = cx + r * Math.cos(-startAngle * rad),
 40                 x2 = cx + r * Math.cos(-endAngle * rad),
 41                 xm = cx + r / 2 * Math.cos(-(startAngle + (endAngle - startAngle) / 2) * rad),
 42                 y1 = cy + r * Math.sin(-startAngle * rad),
 43                 y2 = cy + r * Math.sin(-endAngle * rad),
 44                 ym = cy + r / 2 * Math.sin(-(startAngle + (endAngle - startAngle) / 2) * rad),
 45                 res = ["M", cx, cy, "L", x1, y1, "A", r, r, 0, +(Math.abs(endAngle - startAngle) > 180), 1, x2, y2, "z"];
 46             res.middle = {x: xm, y: ym};  */
 47             //return res;
 48           //if(steps===undefined)
 49             steps=1024;
 50  
 51             startAngle *= Math.PI/180; //convert deg to rad
 52             endAngle *= Math.PI/180;
 53           
 54             var steps1 = Math.floor(parseInt(steps)/3); //steps for lines to tip
 55             var steps2 = steps1 + parseInt(steps) % 3;  //steps for arc
 56             var p1x = cx + r * Math.cos(startAngle);
 57             var p1y = cy - r * Math.sin(startAngle);
 58             var p2x = cx + r * Math.cos(endAngle);
 59             var p2y = cy - r * Math.sin(endAngle);
 60             var angdiff = endAngle - startAngle;
 61       
 62             //create path
 63             var path = [];
 64             path.push('M'+cx+' '+cy); //tip
 65       
 66             //path part 1: tip to arc starting point
 67             var t1x = (p1x-cx)/steps1;
 68             var t1y = (p1y-cy)/steps1; 
 69             
 70             for(var i=1; i<=steps1; i+=1)
 71             {
 72                 path.push('L'+(cx+(t1x*i))+' '+(cy+(t1y*i)));
 73             }
 74           
 75             //path part 2: arc
 76             var tang = angdiff/steps2;
 77             
 78             for(var i=1; i<=steps2; i+=1)
 79             {
 80                 path.push('L'+(cx+r*Math.cos(startAngle+i*tang)));
 81                 path.push(' '+(cy-r*Math.sin(startAngle+i*tang)));
 82             }
 83           
 84             //path part 3: arc end point to tip
 85             var t2x = (cx-p2x)/steps1;
 86             var t2y = (cy-p2y)/steps1; 
 87             
 88             for(var i=1; i<steps1; i+=1)
 89             {
 90                 path.push('L'+(p2x+(t2x*i))+' '+(p2y+(t2y*i)));
 91             }
 92       
 93             //close path
 94             path.push('Z');
 95

101         {
102             total += values[i];

104         }

106         values.sort(function (a, b)
107         {
108             return b.value - a.value;
109         });
110         
111         for (i = 0; i < len; i++)
112         {
113             if (defcut && values[i] * 360 / total <= 1.5)

115                 cut = i;
116                 defcut = false;
117             }
118

121                 defcut = false;
122                 values[cut].value += values[i];

124                 others = values[cut].value;
125             }
126         }
127         
128         len = Math.min(cut + 1, values.length);
129         others && values.splice(len) && (values[cut].others = true);
130               
131         for (i = 0; i < len; i++)
132         {
133             var mangle = angle - 360 * values[i] / total / 2;
134             if (!i)
135             {
136                 angle = 90 - mangle;
137                 mangle = angle - 360 * values[i] / total / 2;
138             }
139             
140             if (opts.init)

143             }

146             //var p = this.path(opts.init ? ipath : path).attr({fill: opts.colors && opts.colors[i] || this.g.colors[i] || "#666", stroke: opts.stroke || "#fff", "stroke-width": (opts.strokewidth == null ? 1 : opts.strokewidth), "stroke-linejoin": "round"});
147             //var p = this.path(path);

154             series.push(pathd);
155             //opts.init && p.animate({path: path.join(",")}, (+opts.init - 1) || 1000, ">");

319 Raphael.fn.g.ringchart = function (cx, cy, r, levels, values, opts)
320 {
321     opts = opts || {};

323         sectors = [],
324         covers = this.set(),
325         chart = this.set(),

328         len = values.length,
329         angle = 0,
330         total = 0,
331         others = 0,
332         cut = 9,
333         defcut = true,
334         radpart = r/levels; //divided radius
335     chart.covers = covers;              
336     
337     
338     if (len == 1)
339     {
340         series.push(this.circle(cx, cy, r).attr({fill: this.g.colors[0], stroke: opts.stroke || "#fff", "stroke-width": opts.strokewidth == null ? 1 : opts.strokewidth}));
341         covers.push(this.circle(cx, cy, r).attr(this.g.shim));
342         total = values[0];
343         values[0] = {value: values[0], order: 0, valueOf: function () { return this.value; }};
344         series[0].middle = {x: cx, y: cy};
345         series[0].mangle = 180;
346     }
347     else
348     {   
349         function ringsector(x, y, rad1, rad2, ang1, ang2, steps)
350         {
351             /* draw an arc segment with given inner radius rad1 and outer radius rad2.
352                start in the middle of the inner arc, go to angle1 (0ï¿½ = east), 
353                go to the outer arc, go ccw to angle2 and back to the inner arc. */
354         
355             if(steps===undefined)
356                 steps=1024;
357          
358             ang1 *= Math.PI/180; //convert deg to rad
359             ang2 *= Math.PI/180;
360             
361             //create variables
362             var steps1 = Math.floor(parseInt(steps)/4); //steps for straight lines and inner arc
363         
364             var steps_inner = Math.floor((rad1 / rad2) * steps1);
365             var steps_outer = steps - 2*steps1 - steps_inner;
366         
367         
368             var steps2 = steps1 + parseInt(steps) % 4;  //steps for outer arc
369             var angdiff = ang2 - ang1;
370             var angdiff2 = angdiff/2;
371             var pi1x = x + rad1 * Math.cos(ang1); //inner circle
372             var pi1y = y - rad1 * Math.sin(ang1);
373             var pi2x = x + rad1 * Math.cos(ang2);
374             var pi2y = y - rad1 * Math.sin(ang2);
375             var pihx = x + rad1 * Math.cos(ang1+angdiff2); //starting point (inner arc center)
376             var pihy = y - rad1 * Math.sin(ang1+angdiff2);
377             var po1x = x + rad2 * Math.cos(ang1); //outer circle
378             var po1y = y - rad2 * Math.sin(ang1);
379             var po2x = x + rad2 * Math.cos(ang2);
380             var po2y = y - rad2 * Math.sin(ang2);
381             
382         
383             console.log(steps1);
384             console.log(steps2);
385         
386             //create path
387             var path = [];
388             path.push('M'+pihx+' '+pihy); //starting point
389         
390             //path part 1: starting point to inner arc end point
391             steps1f = Math.floor(steps_inner/2);
392             var tang = (angdiff2)/steps1f;
393             for(var i=1; i<=steps1f; i+=1) {
394                 path.push('L'+(x+rad1*Math.cos(ang1+angdiff2-i*tang)));
395                 path.push(' '+(y-rad1*Math.sin(ang1+angdiff2-i*tang)));
396             }
397             
398             //path part 2: straight line from inner to outer arc
399             var t1x = (po1x-pi1x)/steps1;
400             var t1y = (po1y-pi1y)/steps1; 
401             for(var i=1; i<=steps1; i+=1) {
402                 path.push('L'+(pi1x+(t1x*i))+' '+(pi1y+(t1y*i)));
403             }
404             
405             //path part 3: outer arc
406             var tang = angdiff/steps_outer;
407             for(var i=1; i<=steps_outer; i+=1) {
408                 path.push('L'+(x+rad2*Math.cos(ang1+i*tang)));
409                 path.push(' '+(y-rad2*Math.sin(ang1+i*tang)));
410             }
411             
412             //path part 4: straight line from outer to inner arc
413             var t2x = (pi2x-po2x)/steps1;
414             var t2y = (pi2y-po2y)/steps1; 
415             for(var i=1; i<=steps1; i+=1) {
416                 path.push('L'+(po2x+(t2x*i))+' '+(po2y+(t2y*i)));
417             }
418             
419             //path part 5: inner arc starting point to starting point
420             steps1c = Math.ceil(steps_inner/2);
421             var tang = angdiff2/steps1c;
422             for(var i=1; i<steps1c; i+=1) {
423                 path.push('L'+(x+rad1*Math.cos(ang2-i*tang)));
424                 path.push(' '+(y-rad1*Math.sin(ang2-i*tang)));
425             }
426         
427             //close path
428             path.push('Z');
429         
430             return path.join('');
431         }
432         //end of ringsector-function
433         
434         for (var i = 0; i < len; i++)
435         {
436             total += values[i][0];
437             values[i][0] = {value: values[i][0], order: i, valueOf: function () { return this.value; }};
438         }
439         
440         values.sort(function (a, b)
441         {
442             return b.value - a.value;
443         });
444         
445         for (i = 0; i < len; i++)
446         {
447             if (defcut && values[i] * 360 / total <= 1.5)
448             {
449                 cut = i;
450                 defcut = false;
451             }
452             
453             if (i > cut) 
454             {
455                 defcut = false;
456                 values[cut][0].value += values[i][0];
457                 values[cut][0].others = true;
458                 others = values[cut].value;
459             }
460         }
461         
462         len = Math.min(cut + 1, values.length);
463         others && values.splice(len) && (values[cut][0].others = true);
464               
465         for (i = 0; i < len; i++)
466         {
467             var mangle = angle - 360 * values[i][0] / total / 2;
468             if (!i)
469             {
470                 angle = 90 - mangle;
471                 mangle = angle - 360 * values[i][0] / total / 2;
472             }
473             
474             if (opts.init)
475             {
476                 var ipath = sector(cx, cy, 1, angle, angle - 360 * values[i] / total).join(",");
477             }
478             
479             var pathd;  //path object
480             var ir, or; //inner and outer radius
481             
482             //value[x] consists of value itself and level
483             /*if(values[i][1]==0)
484             {
485                 pathd = sector(cx, cy, r, angle, angle -= 360 * values[i] / total);
486             }   */

489             or = radpart*(values[i][1]+1);
490                     
491             pathd = ringsector(cx, cy, ir, or, angle, angle -= 360 * values[i][0] / total);
492             
493             //var p = this.path(opts.init ? ipath : path).attr({fill: opts.colors && opts.colors[i] || this.g.colors[i] || "#666", stroke: opts.stroke || "#fff", "stroke-width": (opts.strokewidth == null ? 1 : opts.strokewidth), "stroke-linejoin": "round"});
494             //var p = this.path(path);

497             //p.value = values[i];
